198  def map_to_sphere(self, _v2D):
199  _v3D = [0.0, 0.0, 0.0]
200  # inside Widget?
201  if (( _v2D.x() >= 0 ) and ( _v2D.x() <= self.width() ) and
202  ( _v2D.y() >= 0 ) and ( _v2D.y() <= self.height() ) ):
203  # map Qt Coordinates to the centered unit square [-0.5..0.5]x[-0.5..0.5]
204  x = float( _v2D.x() - 0.5 * self.width()) / self.width()
205  y = float( 0.5 * self.height() - _v2D.y()) / self.height()
206 
207  _v3D[0] = x;
208  _v3D[1] = y;
209  # use Pythagoras to comp z-coord (the sphere has radius sqrt(2.0*0.5*0.5))
210  z2 = 2.0*0.5*0.5-x*x-y*y;
211  # numerical robust sqrt
212  _v3D[2] = math.sqrt(max( z2, 0.0 ))
213 
214  # normalize direction to unit sphere
215  n = linalg.norm(_v3D)
216  _v3D = numpy.array(_v3D) / n
217 
218  return True, _v3D
219  else:
220  return False, _v3D
