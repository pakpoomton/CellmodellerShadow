12 def unique_stable(ar, return_index=False, return_inverse=False):
13  """
14  Find the unique elements of an array.
15 
16  Copied from numpy's unique, but uses stable sorts.
17  """
18  import numpy as np
19  try:
20  ar = ar.flatten()
21  except AttributeError:
22  if not return_inverse and not return_index:
23  items = sorted(set(ar)) # sorted is stable
24  return np.asarray(items)
25  else:
26  ar = np.asanyarray(ar).flatten()
27 
28  if ar.size == 0:
29  if return_inverse and return_index:
30  return ar, np.empty(0, np.bool), np.empty(0, np.bool)
31  elif return_inverse or return_index:
32  return ar, np.empty(0, np.bool)
33  else:
34  return ar
35 
36  if return_inverse or return_index:
37  perm = ar.argsort(kind='mergesort') # using mergesort for stability
38  aux = ar[perm]
39  flag = np.concatenate(([True], aux[1:] != aux[:-1]))
40  if return_inverse:
41  iflag = np.cumsum(flag) - 1
42  iperm = perm.argsort(kind='mergesort') # using mergesort for stability
43  if return_index:
44  return aux[flag], perm[flag], iflag[iperm]
45  else:
46  return aux[flag], iflag[iperm]
47  else:
48  return aux[flag], perm[flag]
49 
50  else:
51  ar.sort(kind='mergesort')
52  flag = np.concatenate(([True], ar[1:] != ar[:-1]))
53  return ar[flag]
